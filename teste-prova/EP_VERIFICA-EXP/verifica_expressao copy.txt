#include <stdio.h>
#include <stdlib.h>
#include "pilha.h"
#include "fila.h"

int main() {

    //char* exp = malloc(101*sizeof(char));
    char exp[101];
    //printf("digite a express√£o\n");
    fgets(exp,101,stdin);
    int i=0;
    int controle = 0;
    char auxDesempilhar;
    
    
    PilhaDinamica *pilha = criar_pilha();

    while (exp[i] != '\0')
    {
        switch (exp[i])
        {
        case '[':
            empilhar(pilha, exp[i]);
            controle++;
            break;
        case '(':
            empilhar(pilha, exp[i]);
            controle++;
            break;
        case '{':
            empilhar(pilha, exp[i]);
            controle++;
            break;
        default:
            if(exp[i] == ']'|| exp[i] == ')' || exp[i] == '}'){
                desempilhar(pilha, &auxDesempilhar);
                controle--;
                if(exp[i] == ']'){
                    if(auxDesempilhar != '[' ){
                        printf("NAO");
                        return 0;
                    }
                }
                else if(exp[i] == ')'){
                    if(auxDesempilhar != '(' ){
                        printf("NAO");
                        return 0;
                    }
                }
                else{
                    if(auxDesempilhar != '{' ){
                        printf("NAO");
                        return 0;
                    }
                }
            }
            break;
        }
       i++;
    }
    //printf("Controle %d", controle);
   if(controle == 0){
       printf("SIM");
   }
   else{
       printf("NAO");
   }

    
    return 0;
}
